#include<bits/stdc++.h>
using namespace std;
int main(){
	int N;
	cin>>N;
	if(N<4||N==6||N==7||N==11){
		cout<<0;
		return 0;
	}
	int ans=0;
	for(int i=0;i<=N/4;i++){
		if((N-4*i)%5==0)ans++;
	}
	cout<<ans;
	return 0;
}
