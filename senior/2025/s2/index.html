#include<bits/stdc++.h>
using namespace std;
typedef unsigned long long ll;
int main() {
    string S;
    ll c;
    cin>>S>>c;
    vector<char> lstChar;
    vector<ll> lstCount;
    string num = "";
    for (char x:S) {
        if ((int)x<48||(int)x>57) {
            if (!num.empty()) {
            	ll pq=0;
            	ll jin=1;
            	for(int i=num.length()-1;i>=0;i--){
            		pq+=jin*((int)num[i]-48);
            		jin*=10;
				}
				if(pq==0){
					lstChar.pop_back();
					
				}
                else lstCount.push_back(pq);
                num="";
            }
            lstChar.push_back(x);
        } else {
            num += x;
        }
    }
	if (!num.empty()) {

        ll pq=0;
        ll jin=1;
        for(int i=num.length()-1;i>=0;i--){
			pq+=jin*((int)num[i]-48);
    		jin*=10;
		}
		lstCount.push_back(pq);
    }
    ll sumOne = 0;
    for (ll n : lstCount) {
        sumOne += n;
    }
    c = (c) % sumOne;
    c+=1;
    ll i = 0;
    while (true) {
        if (c <= lstCount[i]) {
            cout << lstChar[i];
            return 0;
        }
        c -= lstCount[i];
        i++;
    }
    return 0;
}
