#include<bits/stdc++.h>
using namespace std;
vector<vector<long long>> f;
int N;
long long search(int p){
	long long t=0;
	for(int i=0;i<N;i++){
		if(abs(p-f[i][0])-f[i][2]>0)
			t+=(abs(p-f[i][0])-f[i][2])*f[i][1];
	}
	return t;
}
long long md=-1;
int main(){
	cin>>N;
	f.resize(N,vector<long long>(3));
	
	for(int i=0;i<N;i++){
		cin>>f[i][0]>>f[i][1]>>f[i][2];
		md=max(md,f[i][0]);
	}
	int l=0;
	int r=md;
	while(l<=r){
		int mid=(l+r)/2;
		long long lt,mt,rt;
		lt=search(mid-1);
		mt=search(mid);
		rt=search(mid+1);
		if(lt>=mt&&mt<=rt){
			cout<<mt;
			return 0;
		}
		if(lt>=mt&&mt>=rt){
			l=mid+1;
		}
		if(lt<=mt&&mt<=rt){
			r=mid-1;
		}
	}
	return 0;
}
