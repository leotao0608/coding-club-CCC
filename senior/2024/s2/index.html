#include<bits/stdc++.h>
using namespace std;
vector<char> ans;
void check(string str, int index){//'a' 97
	int t[500]={};
	for(int i=0;i<str.length();i++){
		t[(int)str[i]]++;
	}
	int al=0;
	if(t[(int)str[0]]>=2){
		al=1;
	}
	for(int i=1;i<str.length();i++){
		if(t[(int)str[i]]>=2&&al==1){
			ans[index]='F';
			return;
		}
		else if(t[(int)str[i]]>=2&&al==0){
			al=1;
		}else if(t[(int)str[i]]<2&&al==1){
			al=0;
		}else if(t[(int)str[i]]<2&&al==0){
			ans[index]='F';
			return;
		}
	}
	ans[index]='T';
	return;
}
int main(){
	int N,T;
	cin>>T>>N;
	string str;
	ans.resize(T+10);
	for(int i=0;i<T;i++){
		cin>>str;
		check(str,i);
	}
	for(int i=0;i<T;i++){
		cout<<ans[i]<<endl;
	}
	return 0;
}
