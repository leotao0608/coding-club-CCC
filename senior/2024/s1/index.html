#include<bits/stdc++.h>
using namespace std;
int main(){
	int N;
	cin>>N;
	vector<int> H(N);
	for(int i=0;i<N;i++){
		cin>>H[i];
	}
	int ans=0;
	for(size_t i=0;i<H.size()/2;i++){
		if(H[i]==H[N/2+i]){
			ans++;	
		}
	}
	cout<<ans*2;
	return 0;
}
