//score 13/15
#include <bits/stdc++.h>
using namespace std;
int main() {
    int M, N;
    cin >> M >> N;

    vector<vector<int>> grid(M + 1, vector<int>(N + 1));
    for (int i = 1; i <= M; i++) {
        for (int j = 1; j <= N; j++) {
            cin >> grid[i][j];
        }
    }

    queue<pair<int,int>> q;
    vector<vector<bool>> visited(M + 1, vector<bool>(N + 1, false));

    q.push({1, 1});
    visited[1][1] = true;

    while (!q.empty()) {
        auto [r, c] = q.front();
        q.pop();

        if (r == M && c == N) {
            cout << "yes\n";
            return 0;
        }

        int v = grid[r][c];
        for (int a = 1; a * a <= v; a++) {
            if (v % a == 0) {
                int b = v / a;

                if (a <= M && b <= N && !visited[a][b]) {
                    visited[a][b] = true;
                    q.push({a, b});
                }
                if (b <= M && a <= N && !visited[b][a]) {
                    visited[b][a] = true;
                    q.push({b, a});
                }
            }
        }
    }
    cout << "no\n";
    return 0;
}
