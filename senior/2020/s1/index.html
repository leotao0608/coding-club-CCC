#include<bits/stdc++.h>
using namespace std;
int main(){
	int N;
	cin>>N;
	vector<pair<long long,long long>> ts(N);
	for(int i=0;i<N;i++){
		cin>>ts[i].first>>ts[i].second;
	}
	sort(ts.begin(),ts.end());
	double ans=-1;
	for(auto i=0;i<ts.size()-1;i++){
		ans=max(ans,(double)abs(ts[i].second-ts[i+1].second)/(double)abs(ts[i].first-ts[i+1].first));
		
	}
	printf("%.1f",ans);
	return 0;
}
