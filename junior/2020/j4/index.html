#include <bits/stdc++.h>
using namespace std;
int main(){
    string t, s;
    cin>>t>>s;
    for(int i=0;i<s.length();i++){
        s = s.substr(1) + s[0];
        if(t.find(s) != string::npos){  //npos is a return value of function find which indicates not found
            cout<<"yes";
            return 0;
        }
    }
    cout<<"no";
    
    return 0;
}
