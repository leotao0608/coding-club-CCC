#include<bits/stdc++.h>
using namespace std;
int C;
vector<vector<bool>> tile;
vector<vector<bool>> visited;
int n=0,ans=0;
int check(int x, int y){
	if(x<0||x>=2||y<0||y>=C)return 0;
	if(tile[x][y])return 1;
	return 0;
}
void dfs(int x, int y){
	if(x>=0&&x<2&&y>=0&&y<C&&!visited[x][y]&&tile[x][y]){
		visited[x][y]=1;
		
		if(y%2==1) n+=3-check(x,y+1)-check(x,y-1);
		else n+=3-check(x,y+1)-check(x,y-1)-check(x+1,y)-check(x-1,y);
		dfs(x+1,y);
		dfs(x-1,y);
		dfs(x,y+1);
		dfs(x,y-1);
	}else{
		return;
	}
}
int main(){
	cin>>C;
	tile.resize(2,vector<bool> (C));
	visited.resize(2,vector<bool> (C,false));
	for(int i=0;i<2;i++){
		for(int j=0;j<C;j++){
			bool a;
			cin>>a;
			tile[i][j]=a;
		}
	}
	for(int i=0;i<2;i++){
		for(int j=0;j<C;j++){
			if(tile[i][j]&&!visited[i][j]){
				n=0;
				dfs(i,j);
				ans+=n;
			}
		}
	}
	cout<<ans;
	return 0;
}
