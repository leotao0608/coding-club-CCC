#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    string winner, name;
    int bid, maxBid = -1;
    for (int i = 0; i < N; i++) {
        cin >> name >> bid;
        if (bid > maxBid) {
            maxBid = bid;
            winner = name;
        }
    }
    cout << winner << endl;
    return 0;
}
