Problem Guide

The approach is to maintain six variables storing values for the gold, silver and bronze score as well as a count of the 
number of participants with each of these scores. Taking care to initialize these variables properly, they can then be 
updated as needed given each new score. This approach can be used to pass all three subtasks.

Solution Code


#include <bits/stdc++.h>
using namespace std;

int main() {
    int N;
    cin >> N;
    vector<int> a(N);
    for(int i = 0; i < N; i++){
        cin >> a[i];
    }

    vector<int> b = a;
    sort(b.begin(), b.end(), greater<int>());
    b.erase(unique(b.begin(), b.end()), b.end());

    int bronzeScore = b[2];
    int count = 0;
    for(int i = 0; i < N; i++){
        if(a[i] == bronzeScore) count++;
    }

    cout << bronzeScore << " " << count;
    return 0;
}
